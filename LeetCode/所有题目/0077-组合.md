## 最小覆盖子串[leetcode-77](https://leetcode.cn/problems/combinations/)

### 1. 回溯 剪枝
```js
/**
 * @param {number} n
 * @param {number} k
 * @return {number[][]}
 */
var combine = function (n, k) {
  const curr = [];
  const res = [];
  const dfs = (startIndex) => {
    if (curr.length === k) {
      res.push(curr.slice());
      return;
    }
    for (let i = startIndex; i <= n - (k - curr.length) + 1; i++) {
      curr.push(i);
      dfs(i + 1);
      curr.pop();
    }
  };
  dfs(1);
  return res;
};
```
