## 跳跃游戏II[leetcode-45](https://leetcode.cn/problems/jump-game-ii/description/)

### 1. 自己做的
```ts
function jump(nums: number[]): number {
    const len = nums.length;
    let jumpCount = 0; // 跳的总次数
    let jumpIndex = 0; // 当前位置
    // 1. 跳的位置在最后之前开启循环
    while(jumpIndex < len - 1) {
        // 2. 当前跳的格数
        let i = nums[jumpIndex];
        // 3. 可达到末尾，直接返回
        if (jumpIndex + i >= len -1) {
            return jumpCount + 1;
        }
        let temp_max = 0;
        let move = 0;
        // 4. 遍及当下可跳范围中，最远可达的元素
        // - 最远可达即：索引+数值最大
        for (let j = 1; j <= i; j++) {
            if (j + nums[jumpIndex + j] >= move + temp_max) {
                // 5. 找到最大，更新最大者的信息
                temp_max = nums[jumpIndex + j];
                move = j;
            }
        }
        // 6. 将最大者更新为新的弹跳位置
        jumpIndex += move
        jumpCount++;
    }
    return jumpCount;
};
```

### 2. 贪心
```ts
function jump(nums: number[]): number {
    const len = nums.length;
    let maxPositon = 0, end = 0, steps = 0;
    for(let i = 0; i < len - 1; i++) {
        maxPositon = Math.max(maxPositon, i + nums[i])
        if (i === end) {
            end = maxPositon
            steps++;
        }
    }
    return steps
};
```
